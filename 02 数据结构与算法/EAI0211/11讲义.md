# 三维坐标二分查找示例
def search_coordinates(data, target):
    left, right = 0, len(data)-1
    while left <= right:
        mid = (left + right) // 2
        if data[mid] == target:
            return mid
        # 三维坐标比较规则：先x轴，再y轴，最后z轴
        if (data[mid].x, data[mid].y, data[mid].z) < (target.x, target.y, target.z):
            left = mid + 1
        else:
            right = mid - 1
    return -1
